],

],

})
export class AppModule { }
 
app.component.ts:
import { Component } from '@angular/core';
import { enableDebugTools } from '@angular/platform- browser';
import {FormControl,FormControlName,FormGroup, Validators} from '@angular/forms'
@Component({ selector: 'app-root',
templateUrl: './app.component.html', styleUrls: ['./app.component.css']
})
export class AppComponent { title = 'Work With Form'; LoginForm= new FormGroup({
 user:new FormControl('',[Validators.required,Validators.minLength(15)
]),
 password:new FormControl('',[Validators.required,Validators.minLength(6)]
)
})
loginUser(){ console.warn(this.LoginForm.value);
} }
 
App.component.html:
<h1>{{title}}</h1>
<form [formGroup]="LoginForm" (ngSubmit)="loginUser">
<input type="text" placeholder="UserName" name="user" formControlName="user">
 <div class="text-danger"
*ngIf="LoginForm.controls?.user?.touched">
	<span
*ngIf="LoginForm.controls?.user?.errors?.['required']">This is required</span>
</div>
<br><br>
<input type="password" placeholder="password" name="password" formControlName="password">
<div class="text-danger"
*ngIf="LoginForm.controls?.password?.touched">
	<span
*ngIf="LoginForm.controls?.password?.errors?.['required']"> password will be must</span>
</div>
<br><br>
<button>LOGIN</button>
</form>
