    if n==1:
        print("Move disk 1 from",source,"to",destination)
        return
    tower_of_hanoi(n-1,source,destination,auxiliary)
    print("Move disk",n,"From",source,"to",destination)
    tower_of_hanoi(n-1,auxiliary,source,destination)

# DFS using recursion

def dfs(graph, node, visited):
    if node not in visited:
        print(node, end=" ")
        visited.add(node)

        for neighbour in graph[node]:
            dfs(graph, neighbour, visited)

4) Simple BFS Python Program 

def bfs(graph, start):
    visited = []
    queue = [start]

    while queue:
        node = queue.pop(0)
        if node not in visited:
            print(node, end=" ")
            visited.append(node)

            for n in graph[node]:
                queue.append(n)

6) Missionaries & Cannibals – Simple State Space Search 

from collections import deque

def valid(m, c):
    return (m == 0 or m >= c) and (3-m == 0 or 3-m >= 3-c)

def solve():
    start = (3, 3, 0)   # (Missionaries, Cannibals, Boat)
    goal = (0, 0, 1)

    queue = deque([start])
    visited = set()

    while queue:
        m, c, b = queue.popleft()

        if (m, c, b) == goal:
            print("Goal Reached")
            return

        if (m, c, b) in visited:
            continue

        visited.add((m, c, b))

        for dm, dc in [(1,0),(2,0),(0,1),(0,2),(1,1)]:
            if b == 0:
                nm, nc, nb = m-dm, c-dc, 1
            else:
                nm, nc, nb = m+dm, c+dc, 0

            if 0 <= nm <= 3 and 0 <= nc <= 3 and valid(nm, nc):
                queue.append((nm, nc, nb))


solve()
7) Simple Tic Tac Toe using state space search 

# Simple Tic Tac Toe using state space search

board = [' ']*9

def print_board():
    print(board[0], '|', board[1], '|', board[2])
    print('--+---+--')
    print(board[3], '|', board[4], '|', board[5])
    print('--+---+--')
    print(board[6], '|', board[7], '|', board[8])

def check_win(player):
    wins = [(0,1,2),(3,4,5),(6,7,8),
            (0,3,6),(1,4,7),(2,5,8),
            (0,4,8),(2,4,6)]
    for w in wins:
        if board[w[0]] == board[w[1]] == board[w[2]] == player:
            return True
    return False

def minimax(is_ai):
    if check_win('O'):
        return 1
    if check_win('X'):
        return -1
    if ' ' not in board:
        return 0

    best = -2 if is_ai else 2
    for i in range(9):
        if board[i] == ' ':
            board[i] = 'O' if is_ai else 'X'
            score = minimax(not is_ai)
            board[i] = ' '
            best = max(best, score) if is_ai else min(best, score)
    return best

def best_move():
    move = -1
    best = -2
    for i in range(9):
        if board[i] == ' ':
            board[i] = 'O'
            score = minimax(False)
            board[i] = ' '
            if score > best:
                best = score
                move = i
    board[move] = 'O'


# Demo run
print_board()
best_move()
print("\nAI Move:")
print_board()
8) A* Search Algorithm 

def a_star(graph, h, start, goal):
    open_list = [start]
    g = {start: 0}
    parent = {start: None}

    while open_list:
        # Select node with minimum f = g + h
        current = min(open_list, key=lambda x: g[x] + h[x])

        if current == goal:
            path = []
            while current:
                path.append(current)
                current = parent[current]
            print("Path:", path[::-1])
            return

        open_list.remove(current)

        for neighbour, cost in graph[current]:
            new_g = g[current] + cost

            if neighbour not in g or new_g < g[neighbour]:
                g[neighbour] = new_g
                parent[neighbour] = current
                open_list.append(neighbour)


# Graph (node : [(neighbour, cost)])
graph = {
    'A': [('B',1), ('C',3)],
    'B': [('D',1), ('E',4)],
    'C': [('F',5)],
    'D': [],
    'E': [('F',1)],
    'F': []
}

# Heuristic values

a_star(graph, h, 'A', 'F')
9) Travelling Salesman Problem 

import itertools

def tsp(dist):
    cities = range(len(dist))
    min_cost = float('inf')
    best_path = None

    for path in itertools.permutations(cities):
        cost = 0
        for i in range(len(path) - 1):
            cost += dist[path[i]][path[i+1]]
        cost += dist[path[-1]][path[0]]   # return to start

        if cost < min_cost:
            min_cost = cost
            best_path = path

    print("Best Path:", best_path)
    print("Minimum Cost:", min_cost)


# Distance matrix
dist = [
    [0, 10, 15, 20],
    [10, 0, 35, 25],
    [15, 35, 0, 30],
    [20, 25, 30, 0]
]

tsp(dist)
